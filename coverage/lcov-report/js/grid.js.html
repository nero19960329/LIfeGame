var LIVE = 1;
var DEAD = 0;
var WALL = -1;
 
function Grid(width, height) {
	this.width = width;
	this.height = height;
	this.grid = new Array(width);
	for (var i = 0; i < width; ++i) {
		this.grid[i] = new Array(height);
		for (var j = 0; j < height; ++j) {
			this.grid[i][j] = DEAD;
		}
	}
}
